###結構體：

兩種訪問方式

```
struct Student
{
    int sid;
    char name[200]; 
    int age;
}; //分號不能省略
int main(int argc, const char * argv[]) {
    // insert code here...
    struct Student st = {89757, "yanshinian",100};
    
    printf("%d-%s-%d\n", st.sid, st.name, st.age);
    // 第一種方式訪問
    st.sid = 99;
    //st.name = "lisi"; // 這麼寫是錯誤的，數組不能這麼賦值，改成如下
    strcpy(st.name, "lisi"); //#include <string.h>
    st.age = 2;
    printf("\n");
    
    // 第二種方式訪問
    
    struct Student * pst;
    pst = &st;
    pst->sid = 1000; //pst 所指向的結構體變量中的sid這個成員
    printf("%d-%s-%d\n", pst->sid, pst->name, pst->age);
    
    
    return 0;
}
```
注意事項：

 * 結構體變量不能加減乘除，但是可以相互賦值。
 * 普通結構體變量和結構體指針變量作為函數傳參的問題。
 
結構體做為參數傳值：

```
void f(struct Student * pst);
void g(struct Student st);
void g2(struct Student *pst);
int main(int argc, const char * argv[]) {
    struct Student st; // 靜態變量：通過數據類型跟變量名定義的。已經為st分配好了內存
    f(&st);
//     printf("%d-%s-%d\n", st.sid, st.name, st.age);
    g(st);
    g2(&st);
    return 0;
}
void f(struct Student * pst) {
    // 地址賦值給了 pst 而不是 *pst。.*pst等價于st
    (*pst).sid = 99;
    strcpy(pst->name, "xiaoqiang");
    pst->age = 22;
}
// 這種方式耗 內存，耗時間，不推薦 （這裡就能看出指針的好處了吧，速度慢，費空間，因為傳值佔用的空間多大於208個，因為有数据对齐的問題）
void g(struct Student st) {
    printf("%d-%s-%d\n", st.sid, st.name, st.age);
}
// 這個只用了四個字節 （所有的指針變量只佔四個字節）
void g2(struct Student *pst) {
    printf("%d-%s-%d\n", pst->sid, pst->name, pst->age);
}

```
### malloc 動態內存分配和釋放

```
#include <stdlib.h>
int main(int argc, const char * argv[]) {
    // insert code here...
    // 如果沒有使用malloc分配的就是靜態變量。
    
    int a[5] = {4, 10, 2, 8, 6};
    
    int len;
    
    printf("請輸入你需要分配的數組的長度：len＝");
    
    scanf("%d", &len);
    int *pArr = (int *)malloc(sizeof(int) * len);  // malloc請求操作系統為我們分配`sizeof(int) * len`的空間（sizeof(int) * len字節控制權限給我，供我讀寫操作），為什麼加 int * 用來強制轉換呢？malloc只能返回第一個字節的地址，但是第一個字節的地址沒有實際的含義（無意義的地址既干地址），四個字節的int也是用第一個字節表示，八個字節的double也是用第一個字節表示。這就是 int*含義，我們把第一個字節的地址當做整型的地址。pArr ＋ 1指向后四個。如果你不告訴是int的指針，而是double，pArr ＋ 1就指向了后八個
//    *pArr = 4; //類似於 a[0] = 4;
    
//    pArr[1] = 10; // 類似於a[1] = 10;
    // pArr 表示第一個元素的地址，等價于 數組 a
    
//    printf("%d -- %d\n", *pArr, pArr[1]);
    
    // 我們可以把pArr當做一個普通數組來使用
    for (int i = 0; i < len;  ++i) {
        scanf("%d", &pArr[i]);
    }
    for (int i = 0; i<len; ++i) {
        printf("%d\n", *(pArr+i));
    }
    
    free(pArr);// 把pArr所代表的動態內存廢品的20個字節的內存釋放
    
     return 0;
}
```
###跨函數使用內存

要想跨函數的使用內存，只能通過動態的分配內存

```
#include <stdio.h>
#include <malloc/malloc.h>

struct Student {
    int sid;
    int age;
};

struct Student * CreateStudent(void);
void ShowStudent(struct Student *ps);
int main(void) {
    struct Student *ps;
    
    ps = CreateStudent();
    
    ShowStudent(ps);
    
    return 0;
}
struct Student * CreateStudent(void) {
    struct Student *p = (struct Student *)malloc(sizeof(struct Student));
    p->sid = 99;
    p->age = 88;
    return p;
}
void ShowStudent(struct Student *pst) {
    printf("%d--%d\n", pst->sid, pst->age);
}
```

### 鏈表

如何建立一個鏈表呢？首先要一個頭指針header指向鏈表的開始。

```
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

typedef struct Node
{
    int data;// 数据域
    struct Node * pNext; //指针
}NODE, *PNODE;

PNODE create_list(void);
void traverse_list(PNODE head);
bool is_empty(PNODE pHead);
int length_list(PNODE pHead);
bool insert_list(PNODE pHead, int pos, int val);
bool delete_list(PNODE pHead, int pos, int * pVal);
void sort_list(PNODE);

int main(int argc, const char * argv[]) {
    PNODE pHead = NULL; //等价于 struct Node *
    
    pHead = create_list();//创建一个非循环单链表，并将该链表的头结点的地址 赋值给pHead，用于遍历链表之类的
    traverse_list(pHead);
    int len = length_list(pHead);
    printf("len==%d==", len);
    sort_list(pHead);
    printf("排序完成！！！");
    traverse_list(pHead);
    printf("删除完成");
    int  val;
    delete_list(pHead, 3, &val);
    
    traverse_list(pHead);
    return 0;
}

PNODE create_list(void)
{
    int len; //用来存放有效节点的个数
    int i;
    int val; //用来临时存放用户输入的结点的值
    
    PNODE pHead = (PNODE)malloc(sizeof(NODE));
    if (NULL == pHead) {
        printf("分配失败，程序终止！\n");
        exit(-1);
    }
    PNODE pTail = pHead;
    pTail->pNext = NULL;
    printf("请输入您需要生成的链表节点的个数：len=");
    scanf("%d", &len);
    for (i = 0; i < len; ++i) {
        printf("请输入第%d个节点的值：", i+1);
        scanf("%d", &val);
        
        PNODE pNew = (PNODE)malloc(sizeof(NODE));
        if (NULL == pNew) {
            printf("分配失败，程序终止！\n");
            exit(-1);
        }
        pNew->data = val;
        pTail->pNext = pNew;
        pNew->pNext = NULL;
        pTail = pNew;
    }
    return pHead;
}
void traverse_list(PNODE pHead) {
    // 輸出鏈表總的所有數
    PNODE t = pHead->pNext;
    while (NULL != t ) {
        printf("%d\n", t->data);
        t = t->pNext;// 繼續下一個節點
    }
    return;
}
bool is_empty(PNODE pHead)
{
    if (NULL == pHead->pNext) {
        return true;
    } else {
        return false;
    }
}
int length_list(PNODE pHead)
{
    // 輸出鏈表總的所有數
    PNODE t = pHead->pNext;
    int len = 0;
    while (NULL != t ) {
        ++len;
        printf("%d\n", t->data);
        t = t->pNext;// 繼續下一個節點
    }
    return len;
}
void sort_list(PNODE pHead)
{
    int i, j, t, len;
    PNODE p, q;
    len = length_list(pHead);
    printf("len==%d", len);
    for (i = 0, p = pHead->pNext; i < len - 1; ++i, p = p->pNext) {
        for (j = i+1, q = p->pNext; j < len; ++j, q = q->pNext) {
            if (p->data > q->data) {
                t = p->data;
                p->data = q->data;
                q->data = t;
            }
        }
    }
}
// 在pHead 所指向链表的第pos个节点的前面插入一个新的节点。该节点的值为val，pos是插入位置
bool insert_list(PNODE pHead, int pos, int val)
{
    int i = 0;
    PNODE p = pHead;
//    6 7
    while (NULL != p && i < pos-1) {
        p = p->pNext;
        ++i;
    }
    if (i > pos - 1 || NULL == p) {
        return false;
    }
    
    PNODE pNew = (PNODE)malloc(sizeof(NODE));
    if (NULL == pNew) {
        printf("动态分配内存失败!\n");
        exit(-1);
    }
    pNew->data = val;
    PNODE q = p->pNext;
    p->pNext = pNew;
    pNew->pNext = q;
    return true;
}

bool delete_list(PNODE pHead, int pos, int *pVal)
{
    int i = 0;
    PNODE p = pHead;
    //    6 7
    while (NULL != p && i < pos-1) {
        p = p->pNext;
        ++i;
    }
    if (i > pos - 1 || NULL == p) {
        return false;
    }
    
    PNODE q = p->pNext;
    *pVal = q->data;
    
    //删除p节点后面的结点
    
    p->pNext = p->pNext->pNext;
    free(q);
    q = NULL;
    return true;
}
```
